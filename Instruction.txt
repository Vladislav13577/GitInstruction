git status - проверить статус гита
git add -A - добавить все файлы к отслеживанию
git commit -m "Сообщение для фиксации состояния" - фиксирует изменения текущей ветки
git push - отправляет изменения на сайт
git branch - показывает все ветки и на какой сейчас вы
git branch Имя_Ветки - Создаёт новую ветку
git checkout Имя_Ветки - переключает на указанную ветку
git merge Имя_Ветки - соединяет ветку с текущей
git log - отображает историю репозитория
